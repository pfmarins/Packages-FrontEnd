<!---------------------------------- TAG META NAME="ROBOTS" content="noindex" ----------------------------------------->
<!-- Define se uma página deve ou não ser exibida em um buscador (Ex: Google) -->
<!DOCTYPE html>
<html>
    <head>
        <title>Comentários em aberto - DevMedia</title>
         <meta name="robots" content="noindex" />
    </head>
    <body>
        <h1>Relatórios de comentários em aberto</h1>
        <p>Aqui entra o conteúdo do relatório</p>
    </body>
</html>
<!---------------------------------- TAG META NAME="ROBOTS" content="noindex" ----------------------------------------->




<!---------------------------------- TAG META NAME="ROBOTS" content="nofollow" ---------------------------------------->
<!-- Especifica ao buscador para não associar link externos a página que está sendo criada (Ex: Links que são postados em comentários) -->
<!DOCTYPE html>
<html>
    <head>
        <title>Área de comentários</title>
        <meta name="robots" content="nofollow" />
    </head>
    <body>
        <h1>Área de comentários</h1>
        <p>Aqui são exibidos todos os comentários</p>
    </body>
</html>
<!---------------------------------- TAG META NAME="ROBOTS" content="nofollow" ---------------------------------------->
